        function updateColorValues() {
            const hex = colorSelector.value;
            colorDisplay.style.backgroundColor = hex;
            hexValue.textContent = hex.toUpperCase();
            
            // Convert HEX to RGB
            const r = parseInt(hex.substring(1, 3), 16);
            const g = parseInt(hex.substring(3, 5), 16);
            const b = parseInt(hex.substring(5, 7), 16);
            rgbValue.textContent = `rgb(${r}, ${g}, ${b})`;
            
            // Convert RGB to HSL
            const hsl = rgbToHsl(r, g, b);
            hslValue.textContent = `hsl(${Math.round(hsl[0])}, ${Math.round(hsl[1])}%, ${Math.round(hsl[2])}%)`;
        }
        
        function rgbToHsl(r, g, b) {
            r /= 255, g /= 255, b /= 255;
            const max = Math.max(r, g, b), min = Math.min(r, g, b);
            let h, s, l = (max + min) / 2;
            
            if (max === min) {
                h = s = 0; // achromatic
            } else {
                const d = max - min;
                s = l > 0.5 ? d / (2 - max - min) : d / (max + min);
                switch (max) {
                    case r: h = (g - b) / d + (g < b ? 6 : 0); break;
                    case g: h = (b - r) / d + 2; break;
                    case b: h = (r - g) / d + 4; break;
                }
                h *= 60;
            }
            
            return [h, s * 100, l * 100];
        }

        function updateCounts() {
            const text = textToCount.value;
            
            // Word count (split by whitespace and filter out empty strings)
            const words = text.trim() ? text.trim().split(/\s+/).filter(word => word.length > 0) : [];
            wordCount.textContent = words.length;
            
            // Character counts
            characterCount.textContent = text.length;
            characterNoSpaces.textContent = text.replace(/\s+/g, '').length;
            
            // Sentence count (naive approach)
            const sentences = text.split(/[.!?]+/).filter(s => s.trim().length > 0);
            sentenceCount.textContent = sentences.length;
            
            // Paragraph count
            const paragraphs = text.split(/\n+/).filter(p => p.trim().length > 0);
            paragraphCount.textContent = paragraphs.length;
           
 
            // Reading time (average 200 words per minute)
            const minutes = Math.ceil(words.length / 200);
            readingTime.textContent = `${minutes} min`;
            
            // Keyword density (top 5)
            if (words.length > 0) {
                const wordFreq = {};
                words.forEach(word => {
                    const lowerWord = word.toLowerCase().replace(/[^a-z']/g, '');
                    if (lowerWord.length > 3) { // Ignore short words
                        wordFreq[lowerWord] = (wordFreq[lowerWord] || 0) + 1;
                    }
                });
                
                const sortedWords = Object.entries(wordFreq)
                    .sort((a, b) => b[1] - a[1])
                    .slice(0, 5);
                
                keywordDensity.innerHTML = sortedWords.map(([word, count]) => 
                    `
${word}: ${((count / words.length) * 100).toFixed(1)}% (${count})
`
                ).join('');
            } else {
                keywordDensity.innerHTML = '';
            }
            
            countResult.style.display = 'block';
        }

        // Password Strength Checker Tool
        const passwordToCheck = document.getElementById('password-to-check');
        const checkPasswordBtn = document.getElementById('check-password');
        const clearCheckerBtn = document.getElementById('clear-checker');
        const passwordStrengthCheck = document.getElementById('password-strength-check');
        const passwordFeedback = document.getElementById('password-feedback');

        checkPasswordBtn.addEventListener('click', () => {
            const password = passwordToCheck.value;
            if (!password) {
                alert('Please enter a password to check');
                return;
            }
            
            // Calculate password strength
            let strength = 0;
            let feedback = [];
            
            // Length check
            if (password.length >= 12) strength += 2;
            else if (password.length >= 8) strength += 1;
            else feedback.push('Password is too short (minimum 8 characters)');
            
            // Character type checks
            const hasUpper = /[A-Z]/.test(password);
            const hasLower = /[a-z]/.test(password);
            const hasNumber = /[0-9]/.test(password);
            const hasSymbol = /[^A-Za-z0-9]/.test(password);
            
            if (hasUpper) strength += 1;
            else feedback.push('Add uppercase letters');
            
            if (hasLower) strength += 1;
            else feedback.push('Add lowercase letters');
            
            if (hasNumber) strength += 1;
            else feedback.push('Add numbers');
            
            if (hasSymbol) strength += 1;
            else feedback.push('Add symbols');
            
            // Common password check
            const commonPasswords = ['password', '123456', 'qwerty', 'letmein', 'welcome'];
            if (commonPasswords.includes(password.toLowerCase())) {
                strength = 0;
                feedback = ['This is a very common password - choose something more unique'];
            }
            
            // Sequential characters check
            if (/(.)\1{2,}/.test(password)) {
                strength -= 1;
                feedback.push('Avoid repeating characters');
            }
            
            // Determine strength level
            let strengthText, strengthClass;
            if (strength >= 6) {
                strengthText = 'Strong';
                strengthClass = 'strong';
                feedback = ['Great job! This is a strong password'];
            } else if (strength >= 4) {
                strengthText = 'Medium';
                strengthClass = 'medium';
                if (feedback.length === 0) feedback.push('Good start, but could be stronger');
            } else {
                strengthText = 'Weak';
                strengthClass = 'weak';
                if (feedback.length === 0) feedback.push('Very weak password');
            }
            
            passwordStrengthCheck.innerHTML = `Strength: ${strengthText}`;
            passwordFeedback.innerHTML = feedback.map(item => `
• ${item}
`).join('');
            checkerResult.style.display = 'block';
        });
